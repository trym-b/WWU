    is_triggered_only = yes
	
    mean_time_to_happen = {
        days = 1
    }
    
    option = {
        name = wwu_mission_theramore.38.option.a

		#Release Lordaeron as PU
		
		spawn_lordaeron = yes
		
		create_union = A08
		
        ai_chance = {
            factor = 1
        }
    }
	option = {
        name = wwu_mission_theramore.38.option.b      
		
		#Feed it to a Crusader vassal if said vassal exists
		#no immigrants back
		if = {
		    limit = {
			    exists = A22
			}
			spawn_scarlet_lordaeron = yes
		    A22 = {
		        override_country_name = SCARLET_LORDAERON
			    hidden_effect = {
                    set_country_flag = is_scarlet_lordaeron_flag
                }
		    }
		}
		else_if = {
		    limit = {
			    NOT = { exists = A22 } 
			}
			release = A22
			spawn_scarlet_lordaeron = yes
		    A22 = {
		        override_country_name = SCARLET_LORDAERON
			    hidden_effect = {
                    set_country_flag = is_scarlet_lordaeron_flag
                }
		    }
			create_subject = {
                subject_type = vassal
                subject = A22
            }
		}
		
        ai_chance = {
            factor = 1
        }
    }
	option = {
        name = wwu_mission_theramore.38.option.c

		#Keep the lands as they are and bring back immigrants
		add_country_modifier = {
    		name = event_liberation_and_retribution_a23
	    	duration = -1
    	}
		every_owned_province = { 
            limit = {
                region = region_tirisfal_glades
				OR = {
				    culture = culture_scourge
                    culture = culture_forsaken
				}
                NOT = { culture = culture_lordaeron }
            }
            change_culture = culture_lordaeron 
        }
		
        ai_chance = {
            factor = 1
        }
    }
}
