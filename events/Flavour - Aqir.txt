        change_religion = cthun

        add_accepted_culture = culture_qiraji

        add_stability = 1
        add_prestige = 25

        every_owned_province = {
            limit = {
                culture = culture_bronze_dragonflight
            }
            
            change_culture = culture_infinite_dragonflight
        }
        
        if = {
            limit = {
                exists = A97
            }
            
            if = {
                limit = {
                    war_with = A97
                }
                white_peace = A97
            }
            
            add_trust = {
                who = A97
                value = 35
                mutual = yes
            }
            
            create_march = ROOT
        }

    }
    
    after = {
        clear_saved_name = rebel_leader
    }
